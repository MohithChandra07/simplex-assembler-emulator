Program format and structure (Assembler):
=============================

The assembler:
    
    * Written in C++ Language.

    * Every instruction of the SIMPLEX ISA (including the SET pseudo‑instruction) is implemented.

    * It contains a table of instruction names and expected operand types (0 = no operand, 1 = constant, 2 = label/offset).

    * asm.cpp is the main and only file of my assembler.

    * It is a 2‑pass assembler (implemented in pass1() and pass2()).

    * Outputs the required information in three files:
       * .lst – listing file (address, machine code, original assembly line)
       * .log – log file (address, machine code, assembly line, errors/warnings)
       * .o   – object file (binary machine code, 32‑bit words)

    * All important errors are diagnosed along with their line numbers: duplicate label, invalid label name, undefined label, missing operand, unexpected operand, extra text after operand, invalid mnemonic, missing operand for data, SET without label/value, etc.

    * Warnings are generated when a label is declared but never used, and when a data directive is unlabeled.

    * Comments and self‑descriptive variable names have been used.

    * The listing file shows the address and the generated machine code for each instruction/data word. Blank lines and comment lines are skipped and do not increment the program counter (a bug fix is documented in the code).

    * The assembler understands numeric constants in decimal, octal (leading 0) and hexadecimal (0x) formats.

    * The SET instruction is implemented correctly: it assigns the given value to the label without generating any machine code. Forward references are handled because the value is resolved in pass2.

    * The object file is written in binary format, containing only the valid machine code words (excluding SET and error lines).

#include <iostream>
#include <iomanip>
#include <fstream>
#include <sstream>
#include <string>
#include <cstring>
#include <vector>

static const int MEMORY_SIZE = 65536;

enum Reg { REG_A = 0, REG_B = 1, REG_PC = 2, REG_SP = 3 };

struct Emulator {
    int  reg[4];
    int  memory[MEMORY_SIZE];
    bool running;
    int  instruction_count;

    Emulator() : running(true), instruction_count(0) {
        std::memset(reg, 0, sizeof(reg));
        std::memset(memory, 0, sizeof(memory));
    }
};

/* ---------------- HEX FORMAT ---------------- */

static std::string hex8(int v)
{
    std::ostringstream o;
    o << std::hex << std::uppercase
      << std::setfill('0') << std::setw(8)
      << static_cast<unsigned int>(v);
    return o.str();
}

/* ---------------- LOADERS ---------------- */

/* OLD: binary loader (.o) */
static bool load_object(Emulator& emu, const std::string& filename)
{
    std::ifstream file(filename, std::ios::binary);
    if (!file) {
        std::cerr << "Cannot open '" << filename << "'\n";
        return false;
    }

    int addr = 0;
    unsigned int word;

    while (file.read(reinterpret_cast<char*>(&word), sizeof(word))) {
        if (addr >= MEMORY_SIZE) {
            std::cerr << "Object file too large\n";
            return false;
        }
        emu.memory[addr++] = static_cast<int>(word);
    }

    return true;
}

/* NEW: text loader (.mc) */
static bool load_mc_file(Emulator& emu, const std::string& filename)
{
    std::ifstream file(filename);
    if (!file) {
        std::cerr << "Cannot open '" << filename << "'\n";
        return false;
    }

    std::string line;
    int addr = 0;

    while (std::getline(file, line)) {
        if (line.empty()) continue;

        unsigned int word;
        std::stringstream ss;
        ss << std::hex << line;
        ss >> word;

        if (addr >= MEMORY_SIZE) {
            std::cerr << "MC file too large\n";
            return false;
        }

        emu.memory[addr++] = static_cast<int>(word);
    }

    return true;
}

/* ---------------- EXECUTION ---------------- */

static void execute(Emulator& emu)
{
    emu.reg[REG_PC] = 0;
    emu.reg[REG_SP] = 0;

    while (emu.running) {

        int pc = emu.reg[REG_PC];

        if (pc < 0 || pc >= MEMORY_SIZE) {
            std::cerr << "PC out of bounds\n";
            break;
        }

        int instr   = emu.memory[pc];
        int opcode  = instr & 0xFF;
        int operand = (instr >> 8) & 0xFFFFFF;

        /* sign extend */
        if (operand & 0x800000)
            operand |= 0xFF000000;

        int next_pc = pc + 1;

        switch (opcode) {

            case 0: // ldc
                emu.reg[REG_B] = emu.reg[REG_A];
                emu.reg[REG_A] = operand;
                emu.reg[REG_PC] = next_pc;
                break;

            case 1: // adc
                emu.reg[REG_A] += operand;
                emu.reg[REG_PC] = next_pc;
                break;

            case 2: // ldl
                emu.reg[REG_B] = emu.reg[REG_A];
                emu.reg[REG_A] = emu.memory[emu.reg[REG_SP] + operand];
                emu.reg[REG_PC] = next_pc;
                break;

            case 3: // stl
                emu.memory[emu.reg[REG_SP] + operand] = emu.reg[REG_A];
                emu.reg[REG_A] = emu.reg[REG_B];
                emu.reg[REG_PC] = next_pc;
                break;

            case 4: // ldnl
                emu.reg[REG_A] = emu.memory[emu.reg[REG_A] + operand];
                emu.reg[REG_PC] = next_pc;
                break;

            case 5: // stnl
                emu.memory[emu.reg[REG_A] + operand] = emu.reg[REG_B];
                emu.reg[REG_PC] = next_pc;
                break;

            case 6: // add
                emu.reg[REG_A] = emu.reg[REG_B] + emu.reg[REG_A];
                emu.reg[REG_PC] = next_pc;
                break;

            case 7: // sub
                emu.reg[REG_A] = emu.reg[REG_B] - emu.reg[REG_A];
                emu.reg[REG_PC] = next_pc;
                break;

            case 8: // shl
                emu.reg[REG_A] = emu.reg[REG_B] << emu.reg[REG_A];
                emu.reg[REG_PC] = next_pc;
                break;

            case 9: // shr
                emu.reg[REG_A] =
                    static_cast<int>((unsigned)emu.reg[REG_B] >> emu.reg[REG_A]);
                emu.reg[REG_PC] = next_pc;
                break;

            case 10: // adj
                emu.reg[REG_SP] += operand;
                emu.reg[REG_PC] = next_pc;
                break;

            case 11: // a2sp
                emu.reg[REG_SP] = emu.reg[REG_A];
                emu.reg[REG_A] = emu.reg[REG_B];
                emu.reg[REG_PC] = next_pc;
                break;

            case 12: // sp2a
                emu.reg[REG_B] = emu.reg[REG_A];
                emu.reg[REG_A] = emu.reg[REG_SP];
                emu.reg[REG_PC] = next_pc;
                break;

            case 13: // call
                emu.reg[REG_B] = emu.reg[REG_A];
                emu.reg[REG_A] = next_pc;
                emu.reg[REG_PC] = next_pc + operand;
                break;

            case 14: // return
                emu.reg[REG_PC] = emu.reg[REG_A];
                emu.reg[REG_A] = emu.reg[REG_B];
                break;

            case 15: // brz
                emu.reg[REG_PC] =
                    (emu.reg[REG_A] == 0) ? next_pc + operand : next_pc;
                break;

            case 16: // brlz
                emu.reg[REG_PC] =
                    (emu.reg[REG_A] < 0) ? next_pc + operand : next_pc;
                break;

            case 17: // br
                emu.reg[REG_PC] = next_pc + operand;
                break;

            case 18: // HALT
                emu.running = false;
                emu.reg[REG_PC] = next_pc;
                break;

            default:
                std::cerr << "Unknown opcode\n";
                emu.running = false;
        }

        emu.instruction_count++;

        if (emu.instruction_count > 10000000) {
            std::cerr << "Infinite loop suspected\n";
            break;
        }
    }

    std::cout << "\nExecution finished.\n";
    std::cout << "Instructions executed: " << emu.instruction_count << "\n";
    std::cout << "A=" << hex8(emu.reg[REG_A])
              << " B=" << hex8(emu.reg[REG_B])
              << " PC=" << hex8(emu.reg[REG_PC])
              << " SP=" << hex8(emu.reg[REG_SP]) << "\n";
}

/* ---------------- MAIN ---------------- */

int main(int argc, char* argv[])
{
    if (argc < 2) {
        std::cout << "Usage: ./emu <file.o | file.mc>\n";
        return 1;
    }

    std::string filename = argv[1];
    Emulator emu;

    bool loaded = false;

    if (filename.find(".mc") != std::string::npos) {
        loaded = load_mc_file(emu, filename);
    } else {
        loaded = load_object(emu, filename);
    }

    if (loaded) {
        execute(emu);
    }

    return 0;
}
